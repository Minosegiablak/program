/* ════════════════════════════════════════════════

];

const SECTION_MAP = { apps: 'grid-apps', files: 'grid-files', other: 'grid-other' };

/* ── KÁRTYÁK RENDERELÉSE ──────────────────────── */
function renderGrids() {
const saved = getSavedLayout();
Object.entries(SECTION_MAP).forEach(([sec, gridId]) => {
const grid = document.getElementById(gridId);
grid.innerHTML = '';
const ids = saved[sec] || CARDS.filter(c => c.section === sec).map(c => c.id);
ids.forEach(id => {
const baseCard = CARDS.find(c => c.id == id);
if (!baseCard) return;
const overrides = getSavedCards();
const card = overrides[id] ? { ...baseCard, ...overrides[id] } : baseCard;
const el = document.createElement('div');
el.className = 'card';
el.dataset.id = card.id;
el.setAttribute('role', 'button');
el.setAttribute('tabindex', '0');
el.setAttribute('aria-label', card.title);
el.innerHTML = `
${card.isTest ? '<span class="badge-test">Teszt</span>' : ''}
<span class="card-icon">${card.icon}</span>
<span class="card-title">${card.title}</span>
<button class="card-edit-btn" title="Szerkesztés">✎</button>
`;
el.addEventListener('click', (e) => {
if (e.target.classList.contains('card-edit-btn')) { e.stopPropagation(); openEditModal(card); return; }
openCard(card);
});
el.addEventListener('keydown', e => { if (e.key === 'Enter' || e.key === ' ') openCard(card); });
grid.appendChild(el);
});
});
}

/* ── KÁRTYA SZERKESZTÉS ───────────────────────── */
let editingCardId = null;

function getSavedCards() {
try { return JSON.parse(localStorage.getItem('ma_cards')) || {}; } catch { return {}; }
}

function openEditModal(card) {
editingCardId = card.id;
const overrides = getSavedCards();
const data = overrides[card.id] || card;
document.getElementById('edit-title').value = data.title;
document.getElementById('edit-icon').value = data.icon;
document.getElementById('edit-url').value = data.url;
document.getElementById('edit-modal').style.display = 'flex';
}

function closeEditModal() {
document.getElementById('edit-modal').style.display = 'none';
editingCardId = null;
}

function saveCardEdit() {
if (!editingCardId) return;
const overrides = getSavedCards();
const card = CARDS.find(c => c.id == editingCardId);
overrides[editingCardId] = {
...card,
title: document.getElementById('edit-title').value.trim() || card.title,
icon: document.getElementById('edit-icon').value.trim() || card.icon,
url: document.getElementById('edit-url').value.trim() || card.url,
};
localStorage.setItem('ma_cards', JSON.stringify(overrides));
closeEditModal();
renderGrids();
initSortable();
}

document.getElementById('edit-modal').addEventListener('click', e => {
if (e.target === document.getElementById('edit-modal')) closeEditModal();
});

/* ── KÁRTYÁK RENDERELÉSE (override-dal) ───────── */
function openCard(card) {
if (card.isExternal) {
window.open(card.url, '_blank', 'noopener');
} else {
showModal(card.title, card.url);
}
}

function showModal(title, url) {
document.getElementById('modal-title-el').textContent = title;
document.getElementById('modal-url-el').textContent = url;
document.getElementById('modal-open-btn').onclick = () => { window.open(url, '_blank', 'noopener'); closeModal(); };
document.getElementById('link-modal').classList.add('open');
}

function closeModal() {
document.getElementById('link-modal').classList.remove('open');
}

document.getElementById('link-modal').addEventListener('click', e => {
if (e.target === document.getElementById('link-modal')) closeModal();
});

/* ── LAYOUT MENTÉS ────────────────────────────── */
function getSavedLayout() {
try {
return JSON.parse(localStorage.getItem('ma_layout')) || {};
} catch { return {}; }
}

function saveLayout() {
const layout = {};
Object.entries(SECTION_MAP).forEach(([sec, gridId]) => {
layout[sec] = Array.from(document.getElementById(gridId).children).map(el => +el.dataset.id);
});
localStorage.setItem('ma_layout', JSON.stringify(layout));
}

/* ── DRAG & DROP ──────────────────────────────── */
function initSortable() {
Object.values(SECTION_MAP).forEach(gridId => {
new Sortable(document.getElementById(gridId), {
group: 'workspace',
animation: 150,
ghostClass: 'sortable-ghost',
dragClass: 'sortable-drag',
onEnd: saveLayout,
});
});
}

/* ── ÓRA ──────────────────────────────────────── */
function updateClock() {
const now = new Date();
document.getElementById('clock').textContent =
now.toLocaleTimeString('hu-HU', { hour: '2-digit', minute: '2-digit' });

const d = new Date(Date.UTC(now.getFullYear(), now.getMonth(), now.getDate()));
d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay() || 7));
const yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
const week = Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
document.getElementById('week-num').textContent = week;
}

document.getElementById('trello-key').value = localStorage.getItem('t_key') || '';
document.getElementById('trello-token').value = localStorage.getItem('t_token') || '';
document.getElementById('trello-lists').value = localStorage.getItem('t_lists') || '';
}
}

function saveTrelloConfig() {
localStorage.setItem('t_key', document.getElementById('trello-key').value.trim());
localStorage.setItem('t_token', document.getElementById('trello-token').value.trim());
localStorage.setItem('t_lists', document.getElementById('trello-lists').value.trim());
document.getElementById('trello-config-panel').style.display = 'none';
initTrelloSelector();
}

async function initTrelloSelector() {
const key = localStorage.getItem('t_key');
const token = localStorage.getItem('t_token');
const listsStr = localStorage.getItem('t_lists') || '';
const selector = document.getElementById('trello-list-selector');
selector.innerHTML = '';
if (!key || !token || !listsStr) return;

for (const id of listsStr.split(',').map(s => s.trim()).filter(Boolean)) {
try {
const res = await fetch(`https://api.trello.com/1/lists/${id}?key=${key}&token=${token}`);
if (!res.ok) continue;
const data = await res.json();
const opt = document.createElement('option');
opt.value = id;
opt.textContent = data.name;
selector.appendChild(opt);
} catch {}
}
if (selector.options.length > 0) fetchTrelloTasks(selector.value);
}

async function fetchTrelloTasks(listId) {
const key = localStorage.getItem('t_key');
const token = localStorage.getItem('t_token');
const container = document.getElementById('trello-container');
if (!key || !token || !listId) return;

container.innerHTML = '<p class="trello-empty">Betöltés…</p>';
try {
const res = await fetch(`https://api.trello.com/1/lists/${listId}/cards?key=${key}&token=${token}`);
if (!res.ok) throw new Error();
const cards = await res.json();
if (cards.length === 0) {
container.innerHTML = '<p class="trello-empty">Nincs kártya ebben a listában.</p>';
return;
}
container.innerHTML = cards.map(c =>
`<a href="${c.shortUrl}" target="_blank" rel="noopener" class="trello-card">${c.name}</a>`
).join('');
} catch {
container.innerHTML = '<p class="trello-empty">Hiba a betöltés során. Ellenőrizd a beállításokat.</p>';
}
}

function changeTrelloList() {
fetchTrelloTasks(document.getElementById('trello-list-selector').value);
}
